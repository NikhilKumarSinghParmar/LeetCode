class Solution {
public:
    vector<vector<int>> minimumAbsDifference(vector<int>& arr) {
        int n = arr.size();
        sort(arr.begin(),arr.end());
        vector<vector<int>> nums;
        int mindiff = INT_MAX;

        for(int i=1;i<n;i++){
            mindiff = min(mindiff, arr[i]-arr[i-1]);
        }

        for(int i=1;i<n;i++){
            if((arr[i] - arr[i-1]) == mindiff){
                nums.push_back({arr[i-1],arr[i]});
            }
        }

        return nums;
    }
};
