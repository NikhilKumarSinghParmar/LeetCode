class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        int prev = -(k+2);
        int n=nums.size();

        for(int i=0;i<n;i++){
            if(nums[i]==1){
                if((i-prev) > k){
                    prev=i;
                }else{
                    // cout <<  prev << " " << i << endl;
                    return false;
                }
            }
        }

        return true;
    }
};
