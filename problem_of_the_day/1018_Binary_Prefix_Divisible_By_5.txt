#define vb vector<bool>

class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& nums) {
        vb ans;
        int n=nums.size();
        int val=0;

        for(int i=0;i<n;i++){
            val <<= 1;

            if(nums[i]){
                val++;
            }
            val %= 5;

            ans.push_back(!(val%5));
        }

        return ans;
    }
};
